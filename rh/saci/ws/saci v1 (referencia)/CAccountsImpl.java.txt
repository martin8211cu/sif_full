    //***********************
    //*  ADD: 	Metodo para agregar cuentas en DB.
    //***********************
    public long add(java.lang.String pListParameters) throws soinisp.ISPException
    {
		long vResult = -1;
		Connection vConnection = null;
		_Statement1 = null;
		
		try
		{
			vConnection = getConnection(_DefaultDBName);
			if(vConnection != null)
			{
				_Statement1 = vConnection.createStatement();
			
				String pCodCuenta = getParameter(pListParameters, "SAI07COD") ;     // Obtiene el codigo [No es requerido siempre].
				String pNomCuenta = getParameter(pListParameters, "SAI07NCT");      // Nombre de la Cuenta.
				String pCuentaPadre = getParameter(pListParameters, "SAI07CPA");    // Indica si es cuenta padre o no.
				String pImpuestos = getParameter(pListParameters, "SAI07IMP");      // Indica si utiliza impuestos.
				String pVendedor = getParameter(pListParameters, "SAI07VEN");       // Indica si es vendedor.
				String pEstado = getParameter(pListParameters, "SAI07EST");             // Indica el estado de la cuenta.
				String pNombre = getParameter(pListParameters, "SAI07NOM");         // Nombre del dueno de la cuenta.
				String pApellido1 = getParameter(pListParameters, "SAI07AP1");
				String pApellido2 = getParameter(pListParameters, "SAI07AP2");
				String pCedula = getParameter(pListParameters, "SAI07CED");             // Cedula del dueno de la cuenta.
				String pDominio = getParameter(pListParameters, "SAI07DOM");          // Dominio.
				String pPaquete = getParameter(pListParameters, "SAI03COD");            // Codigo de Paquete
				String pTipo = getParameter(pListParameters, "SAI35COD");                   // Tipo de Cuenta.
				String pCalendario = getParameter(pListParameters, "SAI37COD");         // Calendario.
				String pPeriodo = getParameter(pListParameters, "SAI21COD");                // Codigo de Periodo.
				String pCategoria = getParameter(pListParameters, "SAI39COD");          // Categoria
				String pPago = getParameter(pListParameters, "SAI34COD");
				String pPais = getParameter(pListParameters, "SAI30COD");
				String pCodCuentaPadre = getParameter(pListParameters, "SAI07COP");        
				
				try
				{
					Hashtable vTablaServicios = new Hashtable();
					byte vEstadoDB = 0;
					byte vEstadoServers = 1;
			
					// Verifica que el nombre de la cuenta no exista en la DB.
					///ResultSet rsConfirmLogin = doQuery("Select * From SAI007 Where SAI07NCT = '" + pNomCuenta + "'");
					//if(!rsConfirmLogin.next())
					//{
					// Obtiene la lista de Servicios que componen el paquete y los almacena en el HashTable.
					ResultSet rsListaServicios = doQuery("Select SAI010.SAI03COD, SAI002.* " +
														 " From SAI010, SAI002 " +
														 " Where SAI010.SAI03COD = " + pPaquete + 
														 " And SAI010.SAI02COD = SAI002.SAI02COD");
					int vCont = 0;
					while(rsListaServicios.next())
					{
						DatosServicios vDatosServicio = new DatosServicios();
						
						vDatosServicio.vCodServicio = rsListaServicios.getLong("SAI02COD");         // Obtiene el Codigo del Servicio.
						vDatosServicio.vKbytes = rsListaServicios.getLong("SAI02ESP");              // Obtiene el Espacio en que ocupa el servicio. (En este caso no aplica)
						vDatosServicio.vTipoServicio = (int)rsListaServicios.getLong("SAI29COD");   // Obtiene el Tipo de Servicio.
						vCont++;
						vTablaServicios.put(new Integer(vCont), vDatosServicio);
					}
					
					// Si el HashTable contiene rows, la recorre para hacer la inclusiones en los servidores de servicios correspondientes.
					if(vTablaServicios.size() > 0)
					{
						try
						{
						for(vCont = 1; vCont <= vTablaServicios.size(); vCont++)
						{
							Integer objCont = new Integer(vCont);
							DatosServicios vDatosServicio = (DatosServicios)vTablaServicios.get(objCont);
							switch(vDatosServicio.vTipoServicio)
							{
							case 1:         // Servicio de Mail.
							{
								// No existe codigo debido a que en el MailServer no existen las inclusiones de grupos.
								break;
							}
							case 2:     // Servicio de AAA.
							{
								// No existe codigo debido a que en el MailServer no existen las inclusiones de grupos.
								//System.out.println("Inicia el acceso del cliente del AAA");                                
								//vDatosServicio.vEstado = Commands.addAccountAAA(pNomCuenta, pNomCuenta, "null");    // Envia la instrucción de agregar una cuenta en el ServerAAA.
								//System.out.println("El resultado del Cliente del AAA es : " + String.valueOf(vDatosServicio.vEstado));
								//vTablaServicios.put(objCont, vDatosServicio);
								break;
							}
							}       // switch.
							// Si algun server fallo, realiza un rollback y no accesa a la DB.
							if(vDatosServicio.vEstado < 0)
							{
							vEstadoServers = 0;
							vResult = vDatosServicio.vEstado;
							break;
							}
						}       // for.
						} catch(Exception e3) {System.out.println("Los Servers fallaron..... : " + e3.getMessage()); vEstadoServers = 0; vResult = -1;}
						
						// Si los servidores cumplieron con el objetivo.
						if(vEstadoServers == 1)     // Si los server respondieron bien agrega la cuenta en la DB.
						{
						try
						{
							System.out.println("Ejecuta el sp_Alta_SAI007..........");
							System.out.println("exec sp_Alta_SAI007 " + ((pCodCuenta == null)?"":"@SAI07COD=" + pCodCuenta + ", ") + 
											 "@SAI07NCT='" + pNomCuenta +
											 "', @SAI07CPA=" + pCuentaPadre + ", @SAI07IMP=" + pImpuestos + ", @SAI07VEN=" + pVendedor + 
											 ", @SAI07EST='" + pEstado + "', @SAI07NOM='" +pNombre + "',@SAI07AP1='" + pApellido1 +
											 "', @SAI07AP2='" + pApellido2 + "', @SAI07CED='" + pCedula + "', @SAI07DOM='" + pDominio +
											 "', @SAI03COD=" + pPaquete + ", @SAI35COD=" + pTipo + ", @SAI37COD=" + pCalendario + 
											 ", @SAI21COD=" + pPeriodo + ", @SAI39COD=" + pCategoria + ", @SAI34COD=" + pPago + 
											 ", @SAI30COD=" + pPais + ((pCodCuentaPadre == null)?"":", @SAI07COP=" + pCodCuentaPadre));
							ResultSet rs = doQuery("exec sp_Alta_SAI007 " + ((pCodCuenta == null)?"":"@SAI07COD=" + pCodCuenta + ", ") +
											 "@SAI07NCT='" + pNomCuenta +
											 "', @SAI07CPA=" + pCuentaPadre + ", @SAI07IMP=" + pImpuestos + ", @SAI07VEN=" + pVendedor + 
											 ", @SAI07EST='" + pEstado + "', @SAI07NOM='" +pNombre + "',@SAI07AP1='" + pApellido1 +
											 "', @SAI07AP2='" + pApellido2 + "', @SAI07CED='" + pCedula + "', @SAI07DOM='" + pDominio +
											 "', @SAI03COD=" + pPaquete + ", @SAI35COD=" + pTipo + ", @SAI37COD=" + pCalendario + 
											 ", @SAI21COD=" + pPeriodo + ", @SAI39COD=" + pCategoria + ", @SAI34COD=" + pPago + 
											 ", @SAI30COD=" + pPais + ((pCodCuentaPadre == null)?"":", @SAI07COP=" + pCodCuentaPadre));
							if(rs.next())
							{
							vResult = rs.getLong(1);
							vEstadoDB = 1;
							}
							else
							{
							vResult = -3;
							vEstadoDB = 0;
							}
						} catch(Exception e2) {System.out.println("El sp_Alta_SAI007 fallo..... : " + e2.getMessage()); vEstadoDB = 0; vResult = -3;}
						}   // if vEstadoServers.
						
						//********************* 
						//*   ROLLBACK 
						//*********************
						if(vEstadoServers == 0 || vEstadoDB == 0)     // Si los server fallaron o fallo el sp_Alta_SAI007.
						{
						for(vCont = 1; vCont <= vTablaServicios.size(); vCont++)
						{
							Integer objCont = new Integer(vCont);
							DatosServicios vDatosServicio = (DatosServicios)vTablaServicios.get(objCont);
							if(vDatosServicio.vEstado == 1)     // Si el servicio fue exitoso entonces aplicar rollback.
							{
							switch(vDatosServicio.vTipoServicio)
							{
								case 1:         // Servicio de Mail.
								{
								// No existe codigo debido a que en el MailServer no existen las inclusiones de grupos.
								break;                                        
								}
								case 2:     // Servicio de AAA.
								{
								// No existe codigo debido a que en el MailServer no existen las inclusiones de grupos.
								//System.out.println("Inicia el RollBack para borrar la cuenta del AAA");                                
								//vDatosServicio.vEstado = Commands.deleteAccountAAA(pNomCuenta);
								//System.out.println("El resultado del RollBack del AAA es : " + String.valueOf(vDatosServicio.vEstado));
								//vTablaServicios.put(objCont, vDatosServicio);
								break;
								}
							}       // switch.
							}   // if vEstado.
						}   // for.
						}       // if ROLLBACK.
					}       // if vTablaServicios.size
					else
					   vResult = -2;  // Indica que el paquete no tiene servicios
					//}       // if(rsConfirmLogin).
					//else
					//    vResult = -10;       // Indica que el nombre de la cuenta ya existe en la DB.
				} 
				catch(Exception e) 
				{
					System.out.println("CAccountsImpl.add() fallo........ : " + e.getMessage()); 
					vResult = -1;
				}
			}   // if(vConnection != null).
			else
			{
			System.out.println("!!! CAccountsImpl.add() failed.....");
			System.out.println("\tERROR : [Couldn't get a Connection Cache.....]");
			if(_Statement1 != null) {try {_Statement1.close(); _Statement1 = null;} catch(Exception e1) {;}}
			if(vConnection != null) {releaseConnection(_DefaultDBName, vConnection);}
			throw new soinisp.ISPException("Couldn't get a Connection Cache.....");
			}
		}
		catch(Exception e)
		{
			System.out.println("!!! CAccountsImpl.add() failed.....");
			System.out.println("\tERROR : [" + e.getMessage() + "]");
			if(_Statement1 != null) {try {_Statement1.close(); _Statement1 = null;} catch(Exception e1) {;}}
			if(vConnection != null) {releaseConnection(_DefaultDBName, vConnection);}
			throw new soinisp.ISPException(e.getMessage());
		}
		finally
		{
			if(_Statement1 != null) {try {_Statement1.close(); _Statement1 = null;} catch(Exception e1) {;}}
			if(vConnection != null) {releaseConnection(_DefaultDBName, vConnection);}
		}
		return vResult;
    }

										{	//Este componente devuelve el consecutivo de la tarea de cumplimiento

										vEjecutarCumplimientoTarea = true;

														    vAccountData.put("Password", (String)vProperties.get("SAI21PAS"));
														    vAccountData.put("KBytes", ((Long)vServiceData.get("SAI02ESP")).toString());
														    vAccountData.put("Domain", vDomain);
														    vProperties.put("STATE", new Long(Commands.deleteAccountMail(vAccountData, ((Long)vProperties.get("SAI02COD")).longValue())));
														    //vProperties.put("STATE", new Long(1));
