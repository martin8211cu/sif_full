			else
			{
				System.out.println("!!! CUserImpl.changePassword() failed.....");
				System.out.println("\tERROR : [Couldn't get a Connection Cache.....]");

	/*******************************************************************************************
    *   CHANGEPASSWORDCISCO:	Metodo para borra el usuario en Cisco con problemas en passwords
	*                           y lo vuelve agregar con nuevos passwords.
	*							Utilizado en Cambio Password por Sobre y Cambio Password por Servicio
	*                           como alternativa si se da un error en Cisco con metodo changePassword 
    ********************************************************************************************/
    public long changePasswordCisco(String pListParameters)  throws soinisp.ISPException
    {
		long vResult = -1;
		Connection vConnection = null;
		_Statement1 = null;

		try
		{
			vConnection = getConnection(_DefaultDBName);

			if(vConnection != null)
			{
				_Statement1 = vConnection.createStatement();

				byte vEstadoServers = 1;
				String vTelefono = null;
				String vMaxConex = "";

				System.out.println("Inicio del changePasswordCisco en Admon()......");
				// Verifica si el login a borrar exista en la Tabla de Usuarios y obtiene el password viejo.
				String sql = "Select SAI01LOG, SAI03NOM, SAI21PAS, SAI02COD, SAI03TIP, SAI01MNS" +
							 " From SAI001,SAI011,SAI007,SAI003,SAI022" +

							 " And SAI022.SAI01COD = SAI001.SAI01COD" + 
							 " And SAI003.SAI03COD = SAI007.SAI03COD";

				ResultSet rsConfirmLogin = doQuery(sql);

				if(rsConfirmLogin.next())

					String vNomPaquete  = rsConfirmLogin.getString("SAI03NOM");       // Nombre del paquete

						ResultSet rsListTelefono = doQuery("Select SAI72TEL" +

						vTelefono = rsListTelefono.getString("SAI72TEL");    // Lista de Telefono
					}


					// Si el el tipo de paquete es VPN se asigna el # max de conexiones para grabar en Cisco
					String vMaxConexAAA = "null";

					if (TipoPaquete == 3)
						vMaxConexAAA = vMaxConex;

					try 
					{
						// Borra el usuario del Servicio de AAA con password malo.
						System.out.println("Borra usuario Servicio de AAA con password viejo.");
						vDatosServicio.vEstado = Commands.deleteUserAAA(vLogin, vOldPassword, vNomPaquete, vDatosServicio.vCodServicio);
						System.out.println("El resultado es : " + String.valueOf(vDatosServicio.vEstado));

						if(vDatosServicio.vEstado > 0)
						{ // Si se borro el usuario del Servicio de AAA se agrega con el nuevo password.
							Ban_Borrra_Ok = true; // el borrado fue exitoso
							System.out.println("Agrega usuario Servicio de AAA con password nuevo.");
							vDatosServicio.vEstado = Commands.addUserAAA(vLogin, pNewPassword, vNomPaquete, vDatosServicio.vCodServicio, vTelefono, vMaxConexAAA);
							System.out.println("El resultado es : " + String.valueOf(vDatosServicio.vEstado));
						}

						// Si fallo el agregar el usuario con password nuevo 
						if(vDatosServicio.vEstado < 0)
						{
							vEstadoServers = 0;
							vResult = vDatosServicio.vEstado;

							if (Ban_Borrra_Ok)
							{  // Si el borrado del usuario con el password viejo fue exitoso
							   // pero fallo la insercion se debe dejar al usuario como estaba (con password malo)	
								System.out.println("Rollback (Agrega usuario Servicio de AAA con password viejo).");
								vDatosServicio.vEstado = Commands.addUserAAA(vLogin, vOldPassword, vNomPaquete, vDatosServicio.vCodServicio, vTelefono, vMaxConexAAA);

						vEstadoServers = 0; 
						vResult = -1;
					} // end try

							} 
							else
							{
								vEstadoDB = -3; 
								System.out.println("Error en BD changePassword():	" + vEstadoDB);
							}
							vResult = vEstadoDB;
						} 
						catch(Exception e) 
						{
							System.out.println("Error changePasswordCisco() en Server DB SACI..... : " +  e.getMessage());
							vResult = -3;
						}
					} // if(vEstadoServers == 1).


					/*********************
					 *     ROLLBACK 	 *
					 *********************/
					if(vEstadoDB < 0)       // Si fallo el Update SAI022.
					{
						try 
						{
							// Borra el usuario del Servicio de AAA con password nuevo.
							System.out.println("Rollback (Borra usuario Servicio de AAA con password nuevo).");
							vDatosServicio.vEstado = Commands.deleteUserAAA(vLogin, pNewPassword, vNomPaquete, vDatosServicio.vCodServicio);
							System.out.println("El resultado de rollback es : " + String.valueOf(vDatosServicio.vEstado));

							if(vDatosServicio.vEstado > 0)
							{ // Si se borro el usuario del Servicio de AAA se agrega con el viejo password.
								System.out.println("Rollback (Agrega usuario Servicio de AAA con password viejo).");
								vDatosServicio.vEstado = Commands.addUserAAA(vLogin, vOldPassword, vNomPaquete, vDatosServicio.vCodServicio,  vTelefono, vMaxConexAAA);
								System.out.println("El resultado de rollback es : " + String.valueOf(vDatosServicio.vEstado));
							}
						}
						catch(Exception e)
						{
							System.out.println("Error changePasswordCisco() en RollBack..... : " + e.getMessage());
						} // end try	
					}   //if(vEstadoDB == 0).
				}       
				else 
				{
				   vResult = -2;
				   System.out.print("Error [" + String.valueOf(vResult) + "] Usuario no existe en la DB.");
				} // if(rsConfirmLogin.next()).
			}   
			else
			{
				System.out.println("!!! CUserImpl.changePasswordCisco() failed.....");
				System.out.println("\tERROR : [Couldn't get a Connection Cache.....]");
				if(_Statement1 != null) {try {_Statement1.close(); _Statement1 = null;} catch(Exception e1) {;}}
				if(vConnection != null) {releaseConnection(_DefaultDBName, vConnection);}
				throw new soinisp.ISPException("Couldn't get a Connection Cache.....");
			} // if(vConnection != null).
		}
		catch(Exception e)
		{
			System.out.println("!!! CUsersImpl.changePasswordCisco() failed.....");
			System.out.println("\tERROR : [" + e.getMessage() + "]");
			if(_Statement1 != null) {try {_Statement1.close(); _Statement1 = null;} catch(Exception e1) {;}}
			if(vConnection != null) {releaseConnection(_DefaultDBName, vConnection);}
