    private boolean existLCPProtocol(String pBlobData)
    {
        return pBlobData.indexOf("protocol=lcp") != -1;
    }

    private String addLCPProtocol(String pBlobData, long pTimeOut)
        throws Exception
    {
        String vResult = null;
        int vPosServicePPP = pBlobData.indexOf("service=ppp");
        if(vPosServicePPP != -1)
        {
            int vPosIniProtocol = pBlobData.indexOf("{", vPosServicePPP);
            if(vPosIniProtocol != -1)
            {
                String vBeforeSubString = pBlobData.substring(0, vPosIniProtocol + 1);
                String vAfterSubString = pBlobData.substring(vPosIniProtocol + 1);
                String vLCPProtocol = " protocol=lcp { set timeout=" + String.valueOf(pTimeOut) + " }";
                pBlobData = vBeforeSubString + vLCPProtocol + vAfterSubString;
                vResult = pBlobData;
            } else
            {
                throw new Exception("Service ppp bad defined, not found {");
            }
        } else
        {
            throw new Exception("Service ppp not found");
        }
        return vResult;
    }
