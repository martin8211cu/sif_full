    public ExecIPlanetCommands(Socket pSocket, Monitor pMonitor)
    {
		try 
		{

			_PathTemporalFiles = vConfigFileServer.getParameter("MailSender Settings", "PathTemporalFiles");
			_Sender = vConfigFileServer.getParameter("MailSender Settings", "Sender");
			start();    // Calls run()
		} 
		catch(Exception e) 
		{
			System.out.println("ExecMailCommands() fallo......: " + e.getMessage());
		}
    }
    public void run() 
    {
		String inLine = null;
		byte vCodCommand = -1;

		try 
		{
			while(true)
			{
				inLine = in.readLine();      // Recibe la instruccion a ejecutar o el fin de una instruccion.
				out.println(inLine);         // Devuelve la confirmacion de la instruccion recibida.

				if (inLine.equalsIgnoreCase("END |"))
					break;
				StringTokenizer vStrToken = new StringTokenizer(inLine,";");  
				inLine = vStrToken.nextToken().trim();
				vCodCommand = getCodCommand(inLine);
				vMonitor.println("Comando : " + inLine);

				if (vCodCommand != 0) 
				{
					switch (vCodCommand) 
					{
						case 1 : {createAccount(); break;}      // createAccount.
						case 2 : {deleteAccount(); break;}      // deleteAccount.
						case 3 : {updateAccount(); break;}      // updateAccount.
						case 4 : {lockAccount();   break;}      // lockAccount.
						case 5 : {unlockAccount(); break;}      // unlockAccount.
						case 6 : {this.sendMail(); break;}		// Execute the method to send a mail.
						default : {out.println("Error :  Instruccion desconocida"); break;}
					}
				}
			}   // while.
		} 
		catch(Exception e) 
		{
			System.out.println("ExecMailCommands.run() fallo.....: " + e.getMessage());
		} 
		finally 
		{
			try 
			{
				vSocket.close();} catch(Exception e) {System.out.println("Error al cerrar el Socket en ExecMailCommands.run()");
			}
		}
    } // end run
    private byte getCodCommand(String pStrCommand) 
    {
		if(pStrCommand.equalsIgnoreCase("createAccount")) return 1;             // createAccount.
		else if(pStrCommand.equalsIgnoreCase("deleteAccount")) return 2;        // deleteAccount.
		else if(pStrCommand.equalsIgnoreCase("updateAccount")) return 3;        // updateAccount.
		else if(pStrCommand.equalsIgnoreCase("lockAccount")) return 4;          // lockAccount.
		else if(pStrCommand.equalsIgnoreCase("unlockAccount")) return 5;        // unlockAccount.
		else if(pStrCommand.equalsIgnoreCase("sendMail")) return 6;				// Send a Mail.
		else return 0;
    }

	/******************************************************************************************
    *   sendMail() :   Permite desbloquear una cuenta utilizando la linea de comando 
	*				   del PostOffice "sendMail".
    *******************************************************************************************/
    private void sendMail()
    {
		System.out.println("******");
		System.out.println("SendMail init");
		System.out.println("******");
		String vResult = "-1"; 

		try 
		{
			Properties vParameters = new Properties();
			String vTemplateName = in.readLine();	// << Recibe el Nombre de la Plantilla de Mail que se desea enviar.
			out.println(vTemplateName);
			vParameters.put("TemplateName", vTemplateName);
			String vListAccounts = in.readLine();	// << Recibe la lista de cuentas a las que se les debe enviar el Mail
			out.println(vListAccounts);
			vParameters.put("ListAccounts", vListAccounts);
			String vSubject = in.readLine();		// << Recibe el Subject del Mail.
			out.println(vSubject);		
			vParameters.put("Subject", vSubject);
			String vListParameters = in.readLine();	// << Recibe la Lista de los Parametros a aplicar.
			out.println(vListParameters);

			// Distribuye todos los parametros recibidos en una forma mas ordena en un Properties.
			Properties vMessageParameters = new Properties();
			StringTokenizer vStrTok_Parameters = new StringTokenizer(vListParameters, "|");

			do
			{
				String vParametersKey = null;
				Properties vMessageSubParameters = new Properties();
				StringTokenizer vStrTok_Values = new StringTokenizer(vStrTok_Parameters.nextToken(), "^");

				do
				{
					String vParameterValues = vStrTok_Values.nextToken();
					int vDivisionIndex = vParameterValues.indexOf("=");
					String vParameterName = vParameterValues.substring(0, vDivisionIndex);
					String vParameterValue = vParameterValues.substring(vDivisionIndex + 1);
					vMessageSubParameters.put(vParameterName, vParameterValue);
					if(vParameterName.equals("Name"))	// Si es el parametro que se utiliza como identificador unico entre los parametros.
						vParametersKey = vParameterValue;
					System.out.println("Parameter [" + vParameterName + "]" + " Value [" + vParameterValue + "]");
					vParameterValues = null;
					vParameterName = null;
					vParameterValue = null;
				} while(vStrTok_Values.hasMoreTokens());

				vMessageParameters.put(vParametersKey, vMessageSubParameters);
				vParametersKey = null;
				vMessageSubParameters = null;
			} while(vStrTok_Parameters.hasMoreTokens());

			vParameters.put("Parameters", vMessageParameters);
			System.out.println("Todos los parametros [Ok]");

			/*
			String vTemplateName = null;	// Nombre del Template que se usa para la contruccion del Mail.
			switch(Integer.valueOf((String)vParameters.get("MailType")).intValue())
			{
			case 1:			// Envio de Mensaje por Creacion de la Nueva Cuenta.
			{
				vTemplateName = "NewAccount";
				break;
			}
			}
			*/

			// Construye el Encabezado del Archivo que se tomara como base para el envio del Mail.
			File vFileTarget = new File(_PathTemporalFiles, vTemplateName + String.valueOf(System.currentTimeMillis()) +  ".tmp"); 
			PrintWriter vTarget = new PrintWriter(new BufferedWriter(new FileWriter(vFileTarget)), true);
			vTarget.println("From: " + _Sender);
			vTarget.println("To: " + (String)vParameters.get("ListAccounts"));
			vTarget.println("Subject: " + (String)vParameters.get("Subject"));
			vTarget.println("Content-type: text/html");
			vTarget.println("Content-Transfer-Encoding: 8bit");

			// Recorre el File Template para la construccion del Mail.
			String vInputLine = null;
			vMessageParameters = (Properties)vParameters.get("Parameters");
			File vFileTemplate = new File(_PathTemplates, vTemplateName + ".tpl");
			BufferedReader vTemplate = new BufferedReader(new FileReader(vFileTemplate));

			do
			{
				vInputLine = vTemplate.readLine();

				if(vInputLine != null)
				{
					StringBuffer vOutputLine = new StringBuffer();
					int vLastPosition = 0;

					while(true)
					{
						//System.out.println(vInputLine);
						int vTagPosition = vInputLine.indexOf(_TagPrefix, vLastPosition);
						//System.out.println(_TagPrefix);

						if(vTagPosition != -1)
						{
							//System.out.println("vTagPosition : " + vTagPosition);
							vOutputLine.append(vInputLine.substring(vLastPosition, vTagPosition));
							int vTagNamePosition = vInputLine.indexOf("TagName=", vTagPosition + _TagPrefix.length());

							if(vTagNamePosition != -1)
							{
								//System.out.println("vTagNamePosition : " + vTagNamePosition);
								//System.out.println("Ojo : " + ("TagName=").length());
								String vTagName = vInputLine.substring(vTagNamePosition + ("TagName=").length(), vInputLine.indexOf(_TagSufix, vTagNamePosition));
								//System.out.println("TagName : [" + vTagName.trim() + "]");

								if(vTagName != null)
								{
									if(vMessageParameters.containsKey(vTagName.trim()))	// Si el el Tag existe en la lista de Tags recibida.
									{
										Properties vMessageSubParameters = (Properties)vMessageParameters.get(vTagName.trim());
										vOutputLine.append((String)vMessageSubParameters.get("Value"));
									}
									else
									{
										vOutputLine.append(vInputLine.substring(vTagPosition, vInputLine.indexOf(_TagSufix, vTagPosition) + _TagSufix.length()));
									}
								}
							}
							vLastPosition = vInputLine.indexOf(_TagSufix, vTagPosition) + _TagSufix.length();
						}
						else
						{
							vOutputLine.append(vInputLine.substring(vLastPosition, vInputLine.length()));
							break;
						}
					}	// while(true).
					vTarget.println(vOutputLine.toString());
				}	// if(vInputLine != null).
			} while(vInputLine != null);
			vTemplate.close();
			vTarget.close();


		// 	Ejecuta el comando para el envio del mail y contruye los parametros de entrada para dicho comando.
			System.out.println(_PathBinCommands + "sendmail " + "-t " + "< " + vFileTarget.getPath());
			java.lang.Process vProcess = java.lang.Runtime.getRuntime().exec(_PathBinCommands + "sendmail " + "-t " + "< " + vFileTarget.getPath());
			//java.lang.Process vProcess = java.lang.Runtime.getRuntime().exec(_PathBinCommands + "sendmail " + "-t ");

		//	String iPlanetCommand = "/export/home/saci/MAILServerSOL/sendmail.saci";
        //  System.out.println("The Command is: " + iPlanetCommand);
        //  vProcess = Runtime.getRuntime().exec(iPlanetCommand);


			PrintWriter vCommandInput = new PrintWriter(new BufferedWriter(new OutputStreamWriter(vProcess.getOutputStream())));
			BufferedReader vFileInput = new BufferedReader(new FileReader(vFileTarget));
			vInputLine = null;

			do
			{
				vInputLine = vFileInput.readLine();
				if(vInputLine != null)
					vCommandInput.println(vInputLine);
				else
					vCommandInput.println(".");
			} while(vInputLine != null);

			vCommandInput.flush();
			vFileInput.close();
			vCommandInput.close();
			// Espera a que se ejecute la instruccion.
			int vReturnValue = vProcess.waitFor();        
			System.out.println("ReturnValue : " + vReturnValue);
			vResult = new java.io.BufferedReader(new java.io.InputStreamReader(vProcess.getInputStream())).readLine();

			if(vResult == null)
				System.out.println("The result of sendmail is null");
			else
				System.out.println("The result of sendmail is " + vResult);

			// Elimina el archivo temporal del mail.
			vFileTarget.delete();
		}
		catch(Exception e)
		{
			System.out.println("!!! ExecMailCommands.sendMail() failed.....");
			System.out.println("\tERROR : [" + e.getMessage() + "]");
		}
		finally
		{
			out.println(vResult);
		}
    } // sendMail()
