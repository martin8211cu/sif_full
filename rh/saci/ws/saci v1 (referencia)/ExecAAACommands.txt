    public void run() 
    {
		String inLine = null;
		byte vCodCommand = -1;

		try 
		{
			while(true)
			{
				inLine = in.readLine();     // Recibe la instruccion a ejecutar o el fin de una instruccion.
				out.println(inLine);

				if (inLine.equalsIgnoreCase("END |"))
					break;

				StringTokenizer vStrToken = new StringTokenizer(inLine,";");  
				inLine = vStrToken.nextToken();
				vCodCommand = getCodCommand(inLine.trim());
				vMonitor.println("Commando : " + inLine);

				if (vCodCommand != 0) 
				{
					switch (vCodCommand) 
					{
						case 1 :  {this.createPackage(); break;}         	// createPackage.
						case 2 :  {this.deletePackage(); break;}         	// deletePackage.
						case 3 :  {this.changePackagePassword(); break;}  	// changePackagePassword.
						case 4 :  {this.changePackageParent(); break;}    	// changePackageParent.
						case 5 :  {this.createUser(); break;}             	// createUser.
						case 6 :  {this.deleteUser(); break;}             	// deleteUser.
						case 7 :  {this.changeUserPassword(); break;}     	// changeUserPassword.
						case 8 :  {this.changeUserParent(); break;}       	// changeUserParent.
						case 9 :  {this.verifyPassword(); break;}			// verifyPassword.
						case 10 : {this.syncPasswords(); break;}      		// syncPasswords.
						case 11 : {this.updateTimeOut(); break;}			// updateTimeOut.
						case 12 : {this.createTemporalUser(); break;}		// createTemporalUser.
						case 13 : {this.changeUserPasswordAdmon(); break;}	// changeUserPasswordAdmon.
						case 14 : {this.updateTelefonoAAA(); break;}		// updateTelefonoAAA.
						case 15 : {this.createUserVPN(); break;}			// createUserVPN.
						default : {out.println("Error :  Instruccion desconocida"); break;}
					}
				}
			}   // while.
		} catch(Exception e) 
		{
			System.out.println("!!! " + _ClassName + ".run() fallo...");
	    	System.out.println("\tERROR : [" + e.getMessage() + "]");
		} 
		finally 
		{
			try 
			{
				vSocket.close();
			} 
			catch(Exception e) 
			{
				System.out.println("Error al cerrar el Socket en ExecAAACommands.run()");
			}
		}
    } // run()
    private byte getCodCommand(String pStrCommand) 
    {
		if(pStrCommand.equalsIgnoreCase("createPackage")) return 1;                     		 // createPackage.
		else if(pStrCommand.equalsIgnoreCase("deletePackage")) return 2;                		 // deletePackage.
		else if(pStrCommand.equalsIgnoreCase("changePackagePassword")) return 3;       			 // changePackagePassword.
		else if(pStrCommand.equalsIgnoreCase("changePackageParent")) return 4;          		 // changePackageParent.
		else if(pStrCommand.equalsIgnoreCase("createUser")) return 5;                   		 // createUser.        
		else if(pStrCommand.equalsIgnoreCase("deleteUser")) return 6;                   		 // deleteUser.        
		else if(pStrCommand.equalsIgnoreCase("changeUserPassword")) return 7;           		 // changeUserPassword.        
		else if(pStrCommand.equalsIgnoreCase("changeUserParent")) return 8;             		 // changeUserParent.
		else if(pStrCommand.equalsIgnoreCase("verifyPassword")) return 9;               		 // verifyPassword.
		else if(pStrCommand.equalsIgnoreCase("syncPasswords")) return 10;               		 // syncPasswords.
		else if(pStrCommand.equalsIgnoreCase("updateTimeOut")) return 11;               		 // updateTimeOut.
		else if(pStrCommand.equalsIgnoreCase("createTemporalUser")) return 12;          		 // createTemporaUser.
        else if(pStrCommand.equalsIgnoreCase("changeUserPasswordAdmon")) return 13; 			 // changeUserPasswordAdmon.
		else if(pStrCommand.equalsIgnoreCase("updateTelefonoAAA")) return 14; 			 		 // updateTelefonoAAA.
		else if(pStrCommand.equalsIgnoreCase("createUserVPN")) return 15;                   	 // createUserVPN.        
		else return 0;
    }
