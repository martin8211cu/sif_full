    //*   ADD : Agrega un paquete en SACI
    //*******************
    public long add(String pListParameters) throws soinisp.ISPException
    {
		long vResult = -1;
		long S01CON = -1;
		Connection vConnection_SACI = null;
		Connection vConnection_SACISIIC = null;
		_Statement1_SACI = null;
		_Statement1_SACISIIC = null;
		
		try
		{
			vConnection_SACI = getConnection(_SACIDBName);
			vConnection_SACISIIC = getConnection(_SACISIICDBName);
			
			if((vConnection_SACI != null) && (vConnection_SACISIIC != null))
			{
				_Statement1_SACI = vConnection_SACI.createStatement();
				_Statement1_SACISIIC = vConnection_SACISIIC.createStatement();
		
				String pCodPaque = ""; 
				String pNomPaque = getParameter(pListParameters,"SAI03NOM");
				String pDesPaque = getParameter(pListParameters,"SAI03DEC");
				String pFechInic = getParameter(pListParameters,"SAI03FIN");
				String pFechTerm = getParameter(pListParameters,"SAI03FEX");
				String pTipo = getParameter(pListParameters,"SAI03TIP");
				String pTranuc = getParameter(pListParameters,"SAI03TRA");
				String pEstado = getParameter(pListParameters,"SAI03ACT");
				String pAproCamb = getParameter(pListParameters,"SAI03APR");
				String pCtaMini = getParameter(pListParameters,"SAI03PCM");
			
				// se instancia al componente CTareasSACI
				soinisp.CTareasSACI vobjTareasSACI = com.soin.isp.tools.Instances.getInstance_CTareasSACI();
			
				byte vEstadoDBSACI = 0;
				byte vEstadoDBSIIC = 0;
			
				System.out.println("Inic addPackage()");
			
				// Se Envia la tarea de Sincronizacion de la tabla SSXCIN al SIIC
				if (_DebugMode) 
					System.out.println("Envia la tarea SincroSIIC con SSXCIN");
					
				S01CON = vobjTareasSACI.SincroSIIC("TABLA=SSXCIN");
				
				if (_DebugMode) 
					System.out.println("El resultado de SincroSIIC con SSXCIN es:   " + S01CON);
		
				if (S01CON > 0) 
				{ // El envio de la tarea al SIIC fue exitoso
					
					if (_DebugMode) 
						System.out.println("El envio de la tarea al SIIC fue exitoso");	
		
					try
					{
						if (_DebugMode) 
							System.out.println("Ejecuta el sp_Alta_SAI003..........");
						
						ResultSet rsSACI = doQuerySACI("exec sp_Alta_SAI003 @SAI03NOM='" + pNomPaque + 
						"',@SAI03DEC='" + pDesPaque + "', @SAI03FIN='" + pFechInic + "', @SAI03FEX='" + pFechTerm + 
						"',@SAI03TRA='" + pTranuc + "',@SAI03TIP=" + pTipo + ", @SAI03ACT=" + pEstado + ", @SAI03APR=" + pAproCamb + ", @SAI03PCM=" + pCtaMini + "");
			
						if(rsSACI.next())
						{
							vResult = rsSACI.getLong(1);
							vEstadoDBSACI = 1;
							pCodPaque = String.valueOf(vResult);
							
							if (_DebugMode) 
								System.out.println("El paquete es es..... : " + pCodPaque);
						}
						else
						{
							vResult = -3;
							vEstadoDBSACI = 0;
						}
						rsSACI.close();
						rsSACI = null;
					} catch(Exception e) {
						System.out.println("sp_Alta_SAI003 fallo..... : " + e.getMessage()); 
						vEstadoDBSACI = 0; 
						
						if (_DebugMode) 
							System.out.println("El vEstadoDBSACI es..... : " + vEstadoDBSACI);
						vResult = -3;
						
						if (_DebugMode) 
							System.out.println("El vResult es..... : " + vResult);
					}
		
					if (vEstadoDBSACI == 1)
					{// Si el Alta a SAI003 tuvo exito
						try
						{
							if (_DebugMode) 
							{
								System.out.println("Ejecuta el sp_Alta_SSXCIN..........");
								System.out.println("El Alta es exec sp_Alta_SSXCIN @CINCAT=" + pCodPaque +
												   ", @SEVCOD='09', @CINDES='" + pNomPaque + "'");
							}
							
							ResultSet rsSIIC = doQuerySIIC("exec sp_Alta_SSXCIN @CINCAT=" + pCodPaque +
							", @SEVCOD='09', @CINDES='" + pNomPaque + "', @TRANUC='" + pTranuc + "', @CINTIP=" + pTipo);
			
							if(rsSIIC.next())
							{
								vResult = rsSIIC.getLong(1);
								vEstadoDBSIIC = 1;
							}
							else
							{
								vResult = -3;
								vEstadoDBSIIC = 0;
							}
							rsSIIC.close();
							rsSIIC=null;
						} catch(Exception e) {
							System.out.println("El sp_Alta_SSXCIN fallo..... : " + e.getMessage()); 
							vEstadoDBSIIC = 0; 
							
							if (_DebugMode) 
								System.out.println("El vEstadoDBSIIC es..... : " + vEstadoDBSIIC);
							vResult = -3;
							
							if (_DebugMode) 
								System.out.println("El vResult es..... : " + vResult);
						}
					} // end if vEstadoBDSACI	
				} // end if (S01CON > 0)
				
			
				//*********************
				//*   ROLLBACK
				//*********************
				if(/*vEstadoServers == 0 || */vEstadoDBSACI == 0 || vEstadoDBSIIC == 0 || S01CON < 0)
				// Si el server fallo o fallo el sp_Alta_SAI003 o fallo el sp_Alta_SSXCIN  o fallo el enviar la tarea al SIIC
				{
					if (_DebugMode) 
						System.out.println("Inicia el ROLLBACK DEL ADD");
			
					/*// Elimina el paquete en AAA.
					System.out.println("Inicia la eliminación del paquete del AAA");
					int EstadoAAA = Commands.deleteAccountAAA(pNomPaque);
					System.out.println("El resultado del Cliente del AAA es : " + String.valueOf(EstadoAAA));*/
			
					if (S01CON > 0) {  // si se realizo el Baja hace rollback
						// si hubo rollback se debe eliminar la tarea que se envio al SIIC
						S01CON = vobjTareasSACI.DeleteTaskSIIC("S01CON= " + S01CON);
						
						if (_DebugMode) 
							System.out.println("El resultado del DeleteTaskSIIC es :" + S01CON);
					}
			
					if (vEstadoDBSACI == 1) {  // si se realizo el Alta hace rollback
						if (_DebugMode) 
						{
							System.out.println("Inicia el baja en la SAI003");
							System.out.println("El baja es: exec sp_Baja_SAI003 @SAI03COD=" + pCodPaque +"");
						}
						int rsSACI = execUpdateSACI("exec sp_Baja_SAI003 @SAI03COD=" + pCodPaque +"");
						
						if (_DebugMode) 
							System.out.println("El resultado del baja en la SAI003 : " + String.valueOf(rsSACI));
					}	
				} // if ROLLBACK.
				else 
					vResult = 0;        // Todo resulto bien...
		
			}   
			else
			{
				System.out.println("CPackagesImpl.add() failed.....");
				System.out.println("\tERROR : [Couldn't get a Connection Cache]");
				if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e) {;}}
				if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e) {;}}
				if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
				if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
				throw new soinisp.ISPException("Couldn't get a Connection Cache.....");
			}  // if((vConnection_SACI != null) && (vConnection_SACISIIC != null)).
		}
		catch(Exception e)
		{
			System.out.println("CPackagesImpl.add() failed.....");
			System.out.println("\tERROR : [" + e.getMessage() + "]");
			if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e1) {;}}
			if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e1) {;}}
			if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
			if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
			throw new soinisp.ISPException(e.getMessage());
		}
		finally
		{
			if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e) {;}}
			if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e) {;}}
			if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
			if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
		}
		return vResult;
    } // end add

    public long delete(String pListParameters) throws soinisp.ISPException
    {
		long vResult = -1,S01CON = -1;
		Connection vConnection_SACI = null;
		Connection vConnection_SACISIIC = null;
		_Statement1_SACI = null;
		_Statement1_SACISIIC = null;
		
		try
		{
			vConnection_SACI = getConnection(_SACIDBName);
			vConnection_SACISIIC = getConnection(_SACISIICDBName);
			if((vConnection_SACI != null) && (vConnection_SACISIIC != null))
			{
			_Statement1_SACI = vConnection_SACI.createStatement();
			_Statement1_SACISIIC = vConnection_SACISIIC.createStatement();
	
			String pCodPaque = getParameter(pListParameters,"SAI03COD");
		
			// se instancia al componente CTareasSACI
			soinisp.CTareasSACI vobjTareasSACI = com.soin.isp.tools.Instances.getInstance_CTareasSACI();
		
			byte vEstadoDBSACI = 0;
			byte vEstadoDBSIIC = 0;
			//byte vEstadoServers = 1;
		
			System.out.println("Inic deletePackage()");
		
			// Verifica si el paquete a eliminar exista en la Tabla de Paquetes de SACI.
			ResultSet rsSAI003 = doQuerySACI("Select SAI03NOM,CONVERT(varchar,SAI03FIN,101) SAI03FIN," + 
							 " CONVERT(varchar,SAI03FEX,101) SAI03FEX, SAI03ACT, SAI03APR," +
							 " SAI03PCM From SAI003 Where SAI03COD =" + pCodPaque + "");
		
			if (rsSAI003.next())
			{// el paquete a eliminar existe en la Tabla de Paquetes de SACI.
		
				// Obtiene la informacion del paquete en SACI para Rollback
				String pNomPaque = rsSAI003.getString("SAI03NOM");
				String pFechInic = rsSAI003.getString("SAI03FIN");
				String pFechTerm = rsSAI003.getString("SAI03FEX");
				String pEstado = rsSAI003.getString("SAI03ACT");

				ResultSet rsSAI007 = doQuerySACI("Select SAI03COD From SAI007 Where SAI03COD =" + pCodPaque + " and SAI07BOR = 0");	
		
				if (!rsSAI007.next())
				{ // El paquete no esta asociado a una cuenta.
					rsSAI007.close();
					rsSAI007 = null;
			
					// Verifica si el paquete a eliminar exista en la Tabla de Paquetes de SACISIIC.
					ResultSet rsSSXCIN = doQuerySIIC("Select SEVCOD, TRANUC, CINCMI, CINCAH, CINHCM," + 
									 " CINECM From SSXCIN Where CINCAT =" + pCodPaque + "");
			
					if (rsSSXCIN.next())
					{// el paquete a eliminar existe en la Tabla de Paquetes de SACISIIC.
			
						// Obtiene la informacion del paquete en SACISIIC para Rollback
						String pSevcod = rsSSXCIN.getString("SEVCOD");
						String pTranuc = rsSSXCIN.getString("TRANUC");
						String pCincmi = rsSSXCIN.getString("CINCMI");
						String pCincah = rsSSXCIN.getString("CINCAH");
						String pCinhcm = rsSSXCIN.getString("CINHCM");
						String pCinecm = rsSSXCIN.getString("CINECM");
						rsSSXCIN.close();
						rsSSXCIN = null;

						System.out.println("Envia la tarea SincroSIIC con SSXCIN");
						S01CON = vobjTareasSACI.SincroSIIC("TABLA=SSXCIN");
						System.out.println("El resultado de SincroSIIC con SSXCIN es:   " + S01CON);
			
						if (S01CON > 0) { // El envio de la tarea al SIIC fue exitoso
							System.out.println("El envio de la tarea al SIIC fue exitoso");	
			
							try
							{
								System.out.println("Ejecuta el sp_Baja_SAI003.........." + pCodPaque);
								System.out.println("el sp_Baja_SAI003..exec sp_Baja_SAI003 @SAI03COD=" + pCodPaque +"");
								int rsSACI = execUpdateSACI("exec sp_Baja_SAI003 @SAI03COD=" + pCodPaque +"");
								System.out.println("El resultado del baja en la SAI003 : " + String.valueOf(rsSACI));
				
								if (rsSACI >= 0)
								{
									vEstadoDBSACI = 1;
									System.out.println("El paquete es es..... : " + pCodPaque);
								}
								else
								{
									vResult = -3;
									vEstadoDBSACI = 0;
								}
							} catch(Exception e) {
								System.out.println("El sp_Baja_SAI003 fallo..... : " + e.getMessage()); 
								vEstadoDBSACI = 0; 
								System.out.println("El vEstadoDBSACI es..... : " + vEstadoDBSACI);
								vResult = -3;
								System.out.println("El vResult es..... : " + vResult);
							}
			
							if (vEstadoDBSACI == 1)
							{// Si el Baja a SAI003 tuvo exito
								try
								{
									System.out.println("Ejecuta el sp_Baja_SSXCIN..........");
									System.out.println("sp_Baja_SSXCIN exec sp_Baja_SSXCIN @CINCAT=" + pCodPaque + 
									", @SEVCOD='" + pSevcod +"'");
									int rsSIIC = execUpdateSIIC("exec sp_Baja_SSXCIN @CINCAT=" + pCodPaque + 
									", @SEVCOD='" + pSevcod +"', @TRANUC='" + pTranuc + "'");
									System.out.println("El resultado del baja en la SSXCIN : " + String.valueOf(rsSIIC));

										vEstadoDBSIIC = 1;
									else
									{
										vResult = -3;
										vEstadoDBSIIC = 0;
									}

									System.out.println("El vEstadoDBSIIC es..... : " + vEstadoDBSIIC);
									vResult = -3;
									System.out.println("El vResult es..... : " + vResult);
								}
							} // end if vEstadoBDSIIC	
						} // end if (S01CON > 0)
			
						//*********************
						//*   ROLLBACK
						//*********************
						if(/*vEstadoServers == 0 || */vEstadoDBSACI == 0 || vEstadoDBSIIC == 0 || S01CON < 0)
						// Si el server fallo o fallo el sp_Baja_SAI003 o fallo el sp_Baja_SSXCIN o fallo el enviar la tarea al SIIC
						{
							System.out.println("Inicia el ROLLBACK DEL DELETE");
				
							if (S01CON > 0) {  // si se realizo el Baja hace rollback
								// si hubo rollback se debe eliminar la tarea que se envio al SIIC
								S01CON = vobjTareasSACI.DeleteTaskSIIC("S01CON= " + S01CON);
								System.out.println("El resultado del DeleteTaskSIIC es :" + S01CON);

							}

							vResult = 0;        // Todo resulto bien...
					}
					else // Indica que el paquete no existe en Tabla de BD SACISIIC.
						vResult = -10;     
				}
				else // Indica que el paquete esta asociado a alguna cuenta.
				vResult = -11;     
			}
			else // Indica que el paquete no existe en Tabla de SACI. 
				vResult = -12;       
	
			}   // if((vConnection_SACI != null) && (vConnection_SACISIIC != null)).
			else
			{
				System.out.println("!!! CPackagesImpl.delete() failed.....");
				System.out.println("\tERROR : [Couldn't get a Connection Cache.....]");
				if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e) {;}}
				if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e) {;}}
				if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
				if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
				throw new soinisp.ISPException("Couldn't get a Connection Cache.....");
			}
		}
		catch(Exception e)
		{
			System.out.println("CPackagesImpl.delete() failed.....");
			System.out.println("\tERROR : [" + e.getMessage() + "]");
			if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e1) {;}}
			if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e1) {;}}
			if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
			if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
			throw new soinisp.ISPException(e.getMessage());
		}
		finally
		{
			if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e) {;}}
			if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e) {;}}
			if(vConnection_SACI != null) {releaseConnection(_SACIDBName, vConnection_SACI);}
			if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
		}
		return vResult;
    } // end delete

				System.out.println("ZAPPA : " + vZAPPA);

				ResultSet rsPaquete = doQuerySACI("Select SAI03NOM From SAI003 Where SAI03COD = " + pCodPaquete);
				
				if(rsPaquete.next())
				{
					String vNomPaquete = rsPaquete.getString("SAI03NOM");
					System.out.println("Inicio del joinServicePackage()......");
					System.out.println("ZAPPA : " + vZAPPA);
					try
					{
						switch(vZAPPA)
						{
							case 1:     // SI ES SERVICIO DE MAIL.
							{
								vEstadoServers = 1;
								break;
							}
							case 2:     // SI ES SERVICIO DE AAA.
							{

									vEstadoServers = 0;
									vResult = EstadoAAA;
								}
								break;
							}
						}
					} catch(Exception e3) {System.out.println("El componente fallo..... : " + e3.getMessage()); 
					  vEstadoServers = 0;

						{

							System.out.println("El vEstadoDBSACI es..... : " + vEstadoDBSACI);
							vResult = -3;
							System.out.println("El vResult es..... : " + vResult);
						}
					} // end if vEstadoServers
			
					//*********************
					//*   ROLLBACK
					//*********************
					if(vEstadoServers == 0 || vEstadoDBSACI == 0)
					// Si el server fallo o fallo el sp_Alta_SAI003 o fallo el sp_Alta_SSXCIN  o fallo el enviar la tarea al SIIC

							{
								break;
							}

						} // switch(vZAPPA)

				} 
				else

					vResult = -2;
					System.out.println("No se encontro el Codigo del Paquete en la Base de Datos SACI");

			}   // if((vConnection_SACI != null) && (vConnection_SACISIIC != null)).

				if(vConnection_SACISIIC != null) {releaseConnection(_SACISIICDBName, vConnection_SACISIIC);}
				throw new soinisp.ISPException("Couldn't get a Connection Cache.....");
			}
		}
		catch(Exception e)
		{
				System.out.println("!!! CPackagesImpl.disjoinService() failed.....");
				System.out.println("\tERROR : [" + e.getMessage() + "]");
				if(_Statement1_SACI != null) {try {_Statement1_SACI.close(); _Statement1_SACI = null;} catch(Exception e1) {;}}
				if(_Statement1_SACISIIC != null) {try {_Statement1_SACISIIC.close(); _Statement1_SACISIIC = null;} catch(Exception e1) {;}}
