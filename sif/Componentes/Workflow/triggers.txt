	create procedure wf_reorder
		@ProcessId numeric(18) as
	declare
		@depth int, @ActivityId numeric(18)
	begin
		select @depth = 2
	
		update WfActivity
		set Ordering = IsStart
		where ProcessId = @ProcessId
	
		set rowcount 1
		while (1=1) begin
			select @ActivityId = a.ActivityId
			from WfActivity a, WfActivity b, WfTransition t
			where a.ProcessId = @ProcessId
			  and b.ProcessId = @ProcessId
			  and a.ActivityId = t.ToActivity
			  and b.ActivityId = t.FromActivity
			  and b.Ordering != 0
			  and a.Ordering = 0
			  order by a.IsFinish desc
			if @@rowcount = 0 break
	
			update WfActivity
			set Ordering = @depth,
				@depth = @depth + 1
			where ProcessId = @ProcessId
			  and ActivityId = @ActivityId
			if @@rowcount = 0 break
	
		end
		set rowcount 0
	end
	go
	
	
	exec sp_procxmode "wf_reorder", "anymode"
	go
